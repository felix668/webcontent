local.js文件

简介

　　local.js 提供调用终端本地的js方法，如：唤起登录，客户端数据请求，页面跳转，分享，toast提示等。依赖jsbridge.js

方法详解

resApis()

描述：android js安全策略,注册native apis
参数：无
说明：ios平台不需要调用

  Local.resApis();

setBackBtn()

描述：处理页面返回逻辑
参数：无

 JSContent.setIsBackToPage(false); //false代表直接返回  //true逐级返回

init()

描述：页面初始化，注册jsbridge，处理页面返回逻辑
参数：无
说明：ios平台不需要调用

  Local.init();

login(reload_url)

描述：跳转到登录页面
参数：
　　reload_url：可选，页面地址，即登录之后要跳转的页面地址。

Local.login() 
Local.login(window.location.href+"&picking=3");

go(url)

描述：普通重定向
参数：
　　url：必需，跳转页地址

 Local.go('contact.html')

openBookShelf()

描述：去书架
参数：
　　无

 
  Local.openBookShelf();
  跳转app书架原理：window.location.href="uniteqqreader://nativepage/client/bookshelf";

goMontharea()

描述：去包月专区
参数：
　无

 
  Local.gomontharea()
  跳转a包月专区原理:window.location.href="uniteqqreader://nativepage/discover/vipzone";

addToShelf(bid)

描述：加入书架
参数：
　书籍bid，一次只能加一本书，若加多本需要循环操作，或者后台操作

 Local.addToShelf(bid)

reqaObj(url, callback, callbackSections, faildCallback, forceReload)

描述：请求客户端数据
参数：
　　url：必需，请求服务器地址
　　callback：必需，请求成功回调函数
　　callbackSections：可为[]，请求时的默认loading提示
　　faildCallback：可选，请求失败的回调函数
　　forceReload：可选，默认不强制刷新

   Local.reqaObj(server() + "pkg160903/doLottery", function(data) {
     }, [], function() {
        Local.showToast("网络异常，请稍候重试");
      }, 1);

pc端测试链接:
　　tf：必需，PC端走ajax请求
　　skey：必需，ubook.qq.com同步登陆后取cookie里的skey
　　timi/uin：必需，ubook.qq.com同步登陆的qq号 

   adr：http://solomotest4.3g.qq.com/book_res/event/act160902/adr/index.html?tf=1&skey=@VcnluTVWo&timi=1922481934
   ios: http://solomotest4.3g.qq.com/book_res/event/act160902/ios/index.html?tf=1&skey=@zB0qHfnGp&uin=1922481934

doCharge(from,isLogin)

描述：充值跳转
参数：
　　from：非必需，来源
　　isLogin：必需，是否登录

Local.showToast("网络异常，请稍候重试");

showToast(msg)

描述：toast提示
参数：
　　msg：必需，提示信息

Local.showToast("网络异常，请稍候重试");

shareTopic(url, imgUrl, title, intro)

描述：app内分享到其他平台
参数：
　　url：分享链接指向的的页面地址
　　imgUrl：分享缩略图
　　title：分享标题
　　intro：分享描述 

   
      var shareObj = {
        url : common.sharefront()+"act170106/com/index.html",
        cover : common.sharefront()+"act170106/com/images/cover.jpg",

      Local.shareTopic(shareObj.url, shareObj.cover, shareObj.title,shareObj.intro);

setIconForShareing(url,cover,title,intro)

描述：设置客户端右上角分享按钮的参数
参数：同分享设置参数


  var shareObj = {
    url :common.sharefront()+"act170106/com/index.html",

aftershare(url,success)

描述：客户端分享成功回调
参数：
　　url：非必需
　　success：非必需

window.afterShare = function(url,success){
  //分享之后点击返回qq阅读后的回调
}

forceLog(act_f, act_b)

描述：数据统计
参数：
　　act_f：必需，pv,uv来源，五位数字，一般直接获取param("act_f"),url+"?act_f=上线时间如，161101"
　　act_b：必需，约定事件名称 

Local.forceLog(param("act_f"),"SSYinit");

openDetailByCode(dataObj)

描述：跳转到native相应页面
参数：
　dataObj：{pagecode:,bid:,title:,positiestr:,negetivestr:,content:} 


      跳转书籍详情页：
        Local.openDetailByCode({
            pagecode : "1001",
            bid : bid
       });

common.js文件

简介

　　common.js 提供配置服务器地址，页面访问地址，处理url、数据统计的方法。

方法详解

server()

描述：返回服务器地址
参数：无
说明：通过调用可以自动切换正式环境和测试环境的服务器访问主地址

common.server() + "pkg161101/pick";
正式环境：http://event.book.qq.com/andmain/pkg161101/pick
测试环境：http://test3.wapmusic.qq.com/andmain/pkg161101/pick

front()

描述：返回当前页面的访问地址
参数：无
说明：通过调用可以自动切换正式环境和测试环境的页面主地址，可以用来设置ios平台上分享页面的url

common.front() + "pkg161101/adr/index.html";
ios正式环境：https://yuedu.reader.qq.com/event/act170106/ios/index.html
测试环境：https://ptsolomo.reader.qq.com/book_res/event/act170106/ios/index.html

sharefront()

描述：用在app设置当前页面的分享地址
参数：无
说明 ：安卓平台分享到其他平台统一使用https协议，此方法会完成url协议转换

common.sharefront() + "pkg161101/adr/index.html";
adr正式环境：http://iyuedu.qq.com/event/act170106/adr/index.html
测试环境：http://solomotest4.3g.qq.com/book_res/event/act170106/adr/index.html

param()

setParam()

reqa(url, callback, errcallback)

描述：ajax异步请求
参数：
　　url：服务器地址
　　callback:请求成功的回调函数
　　errcallback:请求失败的回调函数


reqa(server() + "andmain/pkg160502/init", function(data) {
});

en.js文件

简介

　　en.js 返回平台信息。此文件返回一个对象env

方法详解

env.pt

描述：根据ua返回平台信息，在app外的平台页面用于实现跨平台操作
结果：ios、adr、ubook

根据平台不同打开不同的页面：N.openPage("http://iyuedu.qq.com/event/act161101/"+env.pt+"/index.html?act_f=161111");

env.vw

描述：根据ua返回平台信息
结果：wx、wb、qq、ot

if (env.vw == "wx") {}

sns-v1.js文件

简介

　　sns-v1.js 处理微信、微博、qq分享、检测各平台是否安装app，依赖wxshare.js、http://pub.idqqimg.com/qqmobile/qqapi.js?_bid=152

方法详解

share(shareinfo, front)

描述：设置端外页面二次分享相关参数
参数：
shareinfo:包含分享标题、描述、图片、url的对象
front:非必需,设置文件依赖路径 默认值https://yuedu.reader.qq.com/event/common


var shareobj={
    "cover" : front()+"act161101/com/images/share2.jpg",
    "url" : window.location.href,
    "title" : "今年买买买，就在QQ阅读",
    "desc" : "我在QQ阅读参加双十一秒杀狂欢，超值包好书不断，抢到就是赚到。"
  };
  S.share(shareobj);

描述：判断是否安装QQ阅读app

front:非必需

    }

native-v1.js文件

简介

　　native-v1.js 处理app端外打开app、下载app等与app操作相关的方法，依赖androidScheme.js、iosscheme.js

方法详解

openPage(url, front)

描述：打开app内页面
参数：
url：页面地址
front:非必需


N.openPage("http://iyuedu.qq.com/event/act161101/adr/index.html?act_f=161111");
adr原理：window.location.href = "uniteqqreader://webpage/" + url;

download(front,channelId)

描述：下载app
参数：
front：非必需
channelId:固定，app渠道号


N.downLoad(null, '10024407');

toBookDetail(bid, front)

描述：跳转到app内书籍详情页
参数：
bid：必需，书籍bid
front：非必需


N.toBookDetail("123456");
原理：window.location.href = "uniteqqreader://nativepage/book/detail?bid=12345";

toReadBook(bid, front)

描述：跳转到app内书籍阅读页
参数：
bid：必需，书籍bid
front：非必需


N.toReadBook("123456");
原理：window.location.href = "uniteqqreader://nativepage/client/readepage?bid=12345";
